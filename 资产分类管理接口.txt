增加资产分类列表
${serverUrl}/manage/assetCategory/addCategory.do
	request：
	{
		Integer id，非必须，不建议填写
		String categoryName，必须，重复将报错
		String detail，非必须
	}
	success：
	{
		{
			"status": 0,
			"msg": "添加资产分类信息成功"
		}
	}
	fail:
	{
		若资产Id已经存在：
		{
			"status": 1,
			"msg": "资产分类已存在"
		}，
		其他错误：
		{
			"status": 1,
			"msg": "添加资产分类信息失败"
		}

	}
