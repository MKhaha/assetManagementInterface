增加集团部门列表
${serverUrl}/manage/department/addDepartment.do
	request：
	{
		Integer id，非必须，不建议填写
		String departmentName，必须，重复将报错
		String detail，非必须
	}
	success：
	{
		{
			"status": 0,
			"msg": "添加集团部门信息成功"
		}
	}
	fail:
	{
		若资产Id已经存在：
		{
			"status": 1,
			"msg": "集团部门已存在"
		}，
		其他错误：
		{
			"status": 1,
			"msg": "添加集团部门信息失败"
		}

	}
